


CREATE TABLE `beh`.`friends` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(500) NOT NULL,
  `createdAt` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`)
);


CREATE TABLE `beh`.`works` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `workname` VARCHAR(500) NOT NULL,
  `friend_id` INT NOT NULL,
  `createdAt` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  CONSTRAINT `fk_friend_id`
    FOREIGN KEY (`friend_id`)
    REFERENCES `friends`(`id`)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);


CREATE TABLE `beh`.`cars` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `carname` VARCHAR(500) NOT NULL,
  `friend_id` INT NOT NULL,
  `createdAt` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  CONSTRAINT `fk_friend_id_cars`
    FOREIGN KEY (`friend_id`)
    REFERENCES `friends`(`id`)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);


CREATE TABLE `beh`.`things` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `thingname` VARCHAR(500) NOT NULL,
  PRIMARY KEY (`id`));




CREATE TABLE `beh`.`friendthings` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `friend_id` INT NOT NULL,
  `thing_id` INT NOT NULL,
  `createdAt` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  CONSTRAINT `fk_friend_id_friends`
    FOREIGN KEY (`friend_id`)
    REFERENCES `friends`(`id`)
    ON DELETE CASCADE
    ON UPDATE CASCADE,
  CONSTRAINT `fk_thing_id_things`
    FOREIGN KEY (`thing_id`)
    REFERENCES `things`(`id`)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);
